# Use official .NET 8 SDK for build
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

COPY ["SkinCareTipsAPI.csproj", "./"]
RUN dotnet restore "SkinCareTipsAPI.csproj"

COPY . .
RUN dotnet build "SkinCareTipsAPI.csproj" -c Release -o /app/build
RUN dotnet publish "SkinCareTipsAPI.csproj" -c Release -o /app/publish

# Use ASP.NET 8 runtime image
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app
COPY --from=build /app/publish .
ENTRYPOINT ["dotnet", "SkinCareTipsAPI.dll"]
